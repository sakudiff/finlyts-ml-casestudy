4. Log Returns

Log returns are computed for all four series. These are used directly as features and as the basis for constructing target variables. The use of lag() ensures that each return uses only the closing price already observed.

df <- master %>%
  mutate(
    r_bpi   = log(bpi_price   / lag(bpi_price)),   # BPI: target asset return
    r_bdo   = log(bdo_price   / lag(bdo_price)),   # BDO: sector competitor
    r_psei  = log(psei_price  / lag(psei_price)),  # PSEi: market index
    r_fx    = log(fx_price    / lag(fx_price)),    # USD/PHP: macro variable 1
    r_us10y = log(us10y_price / lag(us10y_price))  # US 10Y: macro variable 2
  )

5. Target Variables

All three prediction horizons are constructed using lead(), which shifts future returns into the row at time t. These columns are the labels for supervised learning and are never used as predictors.

df <- df %>%
  mutate(
    # 1-day ahead log return: r_{t+1}
    target_1d = lead(r_bpi, 1),

    # 3-day cumulative log return: sum of r_{t+1} + r_{t+2} + r_{t+3}
    target_3d = lead(r_bpi, 1) + lead(r_bpi, 2) + lead(r_bpi, 3),

    # 5-day cumulative log return: sum of r_{t+1} through r_{t+5}
    target_5d = lead(r_bpi, 1) + lead(r_bpi, 2) + lead(r_bpi, 3) +
                lead(r_bpi, 4) + lead(r_bpi, 5)
  )

6. Technical Indicators

All indicators are computed from BPI closing prices or returns using only backward-looking windows. TTR functions (SMA, EMA, RSI, MACD, runSD) return vectors of equal length with NA for the initialization window. No forward-looking information is introduced.

6A. Trend Indicators

df <- df %>%
  mutate(
    # Simple Moving Averages of closing price
    ma5  = SMA(bpi_price, n = 5),
    ma10 = SMA(bpi_price, n = 10),
    ma20 = SMA(bpi_price, n = 20),

    # Exponential Moving Averages: alpha = 2/(k+1), initialized with SMA
    ema12 = EMA(bpi_price, n = 12),
    ema26 = EMA(bpi_price, n = 26),

    # MA ratio: captures short-run vs long-run trend divergence
    # Values > 1 indicate upward short-run momentum relative to trend
    ma_ratio = ma5 / ma20
  )

6B. Momentum Indicators

df <- df %>%
  mutate(
    # Lagged log returns: direct autoregressive predictors
    lag1 = lag(r_bpi, 1),
    lag2 = lag(r_bpi, 2),
    lag3 = lag(r_bpi, 3),

    # Rolling momentum: return over the last k days (r_t minus r_{t-k})
    mom5  = r_bpi - lag(r_bpi, 5),
    mom10 = r_bpi - lag(r_bpi, 10)
  )

# RSI(14): computed directly from closing price using TTR's Wilder smoothing
df <- df %>%
  mutate(rsi14 = RSI(bpi_price, n = 14))

# MACD: TTR returns a two-column matrix; extract and bind explicitly
# to avoid alignment issues from operating on a named matrix inside mutate()
macd_matrix <- MACD(df$bpi_price, nFast = 12, nSlow = 26, nSig = 9, maType = "EMA")
df <- df %>%
  mutate(
    macd        = as.numeric(macd_matrix[, "macd"]),
    macd_signal = as.numeric(macd_matrix[, "signal"])
  )

6C. Volatility Indicators

df <- df %>%
  mutate(
    # Rolling sample standard deviation of log returns (biased=FALSE by default)
    vol5  = runSD(r_bpi, n = 5,  sample = TRUE),
    vol10 = runSD(r_bpi, n = 10, sample = TRUE),
    vol20 = runSD(r_bpi, n = 20, sample = TRUE),  # also used for regime variable

    # Short-to-medium-run volatility ratio
    vol_ratio = vol5 / vol10
  )

8. Feature Transformations

Two engineered features capture nonlinear and conditional dynamics that linear predictors cannot exploit on their own.

df <- df %>%
  mutate(
    # Interaction: lagged return × short-run volatility
    # Tests whether momentum signal is stronger/weaker in high-vol environments
    interact = lag1 * vol5,

    # Squared short-run volatility: nonlinear volatility effect
    vol5_sq  = vol5^2
  )

10. Regime Variable

The high-volatility flag threshold (sigma_bar) is computed on the training partition only to prevent look-ahead bias — if computed on the full sample it would embed future volatility information into a historical feature.

# Strict time-series split: first 80% = training, last 20% = test
n_total <- nrow(features)
n_train  <- floor(0.80 * n_total)
n_test   <- n_total - n_train

train_cutoff <- features$date[n_train]

cat("Total clean observations :", n_total, "\n")
## Total clean observations : 1853
cat("Training rows (80%)      :", n_train, " | up to", format(train_cutoff, "%Y-%m-%d"), "\n")
## Training rows (80%)      : 1482  | up to 2024-10-22
cat("Test rows     (20%)      :", n_test,  " | from",
    format(features$date[n_train + 1], "%Y-%m-%d"), "\n")
## Test rows     (20%)      : 371  | from 2024-10-23
# Compute threshold from training rows only
sigma_bar <- mean(features$vol20[1:n_train], na.rm = TRUE)
cat("\nHigh-vol threshold (sigma_bar) :", round(sigma_bar, 6), "\n")
## 
## High-vol threshold (sigma_bar) : 0.018117
# Apply regime flag to all rows (using the training-derived threshold)
features <- features %>%
  mutate(hi_vol = as.integer(vol20 > 2 * sigma_bar))

cat("High-vol days (full sample)    :", sum(features$hi_vol),
    "/", n_total,
    "(", round(100 * mean(features$hi_vol), 1), "%)\n")

12. Descriptive Statistics

Summary statistics for the three target variables and a representative selection of features. Exported to paper/tables/descriptive_stats.tex for direct inclusion in the paper.

# Helper: compute skewness and excess kurtosis
skew <- function(x) {
  x <- x[!is.na(x)]
  mean((x - mean(x))^3) / sd(x)^3
}
exkurt <- function(x) {
  x <- x[!is.na(x)]
  mean((x - mean(x))^4) / sd(x)^4 - 3
}

# Variables to summarize
